# Holy Order Decisions are listed for each existing holy order of the right religion

decisions = {
	
	expel_black_fang = {
		only_independent = yes
		ai = no

		potential = {
			OR = {
				religion = elimine_church
				religion_group = magic_group
			}
			is_heretic = no
			higher_tier_than = DUKE
			independent = yes
			is_title_active = d_black_fang
			NOT = { has_character_modifier = expelled_d_black_fang }
			
			any_realm_province = {
				any_province_lord = {
					d_black_fang = {
						holder_scope = {
							OR = {
								character = PREVPREV
								is_vassal_or_below = PREVPREV
							}
						}
					}
				}
			}
			
			has_dlc = "Sons of Abraham"
			NOT = { society_member_of = black_fang }
		}
		
		allow = {
			piety = 1000
			custom_tooltip = {
				text = expelling_holy_order_decision_tooltip
				hidden_tooltip = {
					NOT = { has_character_flag = expel_black_fang_check }
				}
			}
		}
		
		effect = {
			set_character_flag = expel_black_fang_check
			custom_tooltip = {
				text = expelling_holy_order_decision_tooltip_b
				hidden_tooltip = {
					character_event = { id = SoA.2990 }
				}
			}
		}
		
		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	
	borrow_money_black_fang = {
		only_playable = yes
		ai_check_interval = 120

		potential = {
			is_playable = yes
			OR = {
				religion = elimine_church
				religion_group = magic_group
			}
			is_heretic = no
			is_title_active = d_black_fang
			OR = {
				ai = no
				AND = {
					ai = yes
					higher_tier_than = COUNT
					independent = yes
				}
			}
			NOT = { has_character_modifier = borrowed_from_black_fang }
			NOT = { has_character_modifier = expelled_d_black_fang }
			NOT = { is_nomadic = yes }
			
			has_dlc = "Sons of Abraham"
			NOT = { society_member_of = black_fang }
			NOT = { government = order_government }
			
			d_black_fang = {
				holder_scope = {
					liege = {
						NOT = { character = ROOT }
					}
				}
			}

			NOT = { has_character_modifier = black_fang_debt } #From MNM.6001
		}
		
		allow = {
			d_black_fang = {
				holder_scope = {
					wealth = 300
				}
			}

			piety = 100

			custom_tooltip = {
				text = black_fang_loan_money_TT

				d_black_fang = {
					holder_scope = {
						opinion = {
							who = ROOT
							value = 25
						}
					}
				}
			}
		}
		
		effect = {
			add_character_modifier = {
				name = borrowed_from_black_fang
				duration = -1
				inherit = yes
			}
			
			hidden_tooltip = {
				d_black_fang = {
					holder_scope = {
						character_event = {
							id = SoA.1410
						}
					}
				}
			}
			
			wealth = 300

			piety = -100

			hidden_effect = {
				if = {
					limit = { has_dlc = "Mystics" }
					set_variable = { which = black_fang_debt value = 300 }
				}
			}
		}
		
		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				wealth = 0
			}
			modifier = {
				factor = 0.5 # Slow it down
			}
		}
	}
	
	donate_money_to_black_fang = {
		only_playable = yes
		ai_check_interval = 80

		potential = {
			is_playable = yes
			
			OR = {
				AND = {
					wealth = 1000
					NOR = {
						trait = cynical
						piety = 50
					}
				}
				ai = no
			}
			
			NOT = { government = order_government }

			is_title_active = d_black_fang
			
			d_black_fang = {
				holder_scope = { 
					religion = ROOT
				}
			}
			
			NOT = { has_character_modifier = expelled_d_black_fang }

			NOT = { society_member_of = black_fang }
			
			has_dlc = "Sons of Abraham"
			
			d_black_fang = {
				holder_scope = {
					liege = {
						NOT = { character = ROOT }
					}
				}
			}
		}
		
		allow = {
			wealth = 300
			custom_tooltip = {
				text = holy_order_donate_cooldown_tt
				NOT = { has_character_modifier = donate_to_holy_order_cooldown }
			}
		}
		
		effect = {
			hidden_effect = {
				add_character_modifier = {
					modifier = donate_to_holy_order_cooldown
					hidden = yes
					years = 1
				}
			}
			
			hidden_tooltip = {
				d_black_fang = {
					holder_scope = {
						character_event = {
							id = SoA.1140
						}
					}
				}
			}
			
			wealth = -300
			piety = 100
		}

		revoke_allowed = {
			always = no
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0.1 # Slow it down
			}
		}
	}
}
