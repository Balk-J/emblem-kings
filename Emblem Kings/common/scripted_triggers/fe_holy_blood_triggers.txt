can_inherit_holy_blood_trigger = {
	OR = {
		AND = {
			is_father_real_father = yes
			father_even_if_dead = {
				has_holy_blood_trigger = yes
			}
		}
		AND = {
			is_father_real_father = no
			real_father_even_if_dead = {
				has_holy_blood_trigger = yes
			}
		}
		mother_even_if_dead = {
			has_holy_blood_trigger = yes
		}
	}
}

		has_minor_holy_blood_trigger = yes
