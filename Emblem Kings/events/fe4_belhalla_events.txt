# Door of Destiny event chain - battle of Behalla, Yied Massacre and related events
character_event = {
	id = fe4_belhalla.1
	desc = fe4_belhalla.1.desc
	picture = GFX_evt_rider_adventure
	hide_from = yes
	is_triggered_only = yes
		
	option = {
		name = fe4_belhalla.1.A
		ai_chance = {
			factor = 90	
		}
		# Give an opinion boost to liege so they don't attack
		hidden_tooltip = {
			liege = {
				opinion = {
					modifier = opinion_appreciated_abilities
					who = ROOT
					years = 5
				}				
			}
		}
		
		
		# Give Sigurd the home castle for this war
		if = {
			limit = {
				liege = {
					any_realm_title = { title = c_zaxon }
				}
			}			
			c_zaxon = {
				set_title_flag = battle_of_belhalla_base
				grant_title = ROOT
			}
			liege = {
				ROOT = {
# Doesn't seem to work - new title always gets the default.				
#					primary_title = {
#						d_sigurd = {
#							copy_title_laws = PREV
#						}
#					}
					any_demesne_title = {
						limit = {
							NOT = { title = c_zaxon }
							NOT = { de_jure_liege_or_above = c_zaxon }
						}
						grant_title = PREVPREV
					}
				}
			}
		}else = {
			liege = {
				ROOT = {
					random_demesne_title = {
						limit = {
							NOT = { is_capital = yes }
						}
						set_title_flag = battle_of_belhalla_base
						grant_title = PREVPREV
					}
				}
			}
		}
		set_defacto_liege = THIS		
		d_sigurd = {
			grant_title = ROOT 
			hidden_tooltip = {
				add_law = { law = succ_primogeniture cooldown = no opinion_effect = no }
				add_law = { law = true_cognatic_succession cooldown = no opinion_effect = no }
			}
		}
		character_event = { id = fe4_belhalla.2 }
	}	
	option = {
		name = fe4_belhalla.1.B
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				has_game_rule = {
					name = fe_story
					value = on
				}
			}			
		}
		prestige = -100
	}	
}
# Find the leader of the war against Sigurd
character_event = {
	id = fe4_belhalla.2
	hide_window = yes
	is_triggered_only = yes
		
	option = {
		name = OK
		# Prefer Lombard
		if = {
			limit = {
				c_1070 = {
					is_alive = yes
					independent = no
					liege = {
						has_landed_title = k_grannvale
					}
					has_landed_title = d_dozel
					NOT = { has_character_flag = refused_belhalla_war }
				}
			}
			c_1070 = {
				set_character_flag = battle_of_belhalla_war_leader
			}
		}
		# Then Reptor 
		else_if = {
			limit = {
				c_1056 = {
					is_alive = yes
					independent = no
					liege = {
						has_landed_title = k_grannvale
					}
					has_landed_title = d_friege
					NOT = { has_character_flag = refused_belhalla_war }
				}
			}
			c_1056 = {
				set_character_flag = battle_of_belhalla_war_leader
			}
		}
		# Then try a Dozel who isn't Lex
		else_if = {
			limit = {
				d_dozel = {
					holder_scope = {
						independent = no
						liege = {
							has_landed_title = k_grannvale
						}
						dynasty = 1006 # Dozel
						NOT = { has_character_flag = is_fe4_lex }
						NOT = { has_character_flag = refused_belhalla_war }
					}
				}
			}
			d_dozel = {
				holder_scope = {
					set_character_flag = battle_of_belhalla_war_leader
				}
			}
		}
		# Then try a Friege who isn't one of the good sisters
		else_if = {
			limit = {
				d_friege = {
					holder_scope = {
						independent = no
						liege = {
							has_landed_title = k_grannvale
						}
						dynasty = 1005 # Friege
						NOT = { has_character_flag = is_fe4_tailtiu }
						NOT = { has_character_flag = is_fe4_ethnia }
						NOT = { has_character_flag = refused_belhalla_war }
					}
				}
			}
			d_friege = {
				holder_scope = {
					set_character_flag = battle_of_belhalla_war_leader
				}
			}
		}
		# Then pick a random powerful vassal who isn't going to be on Sigurd's side
		else = {
			k_grannvale = {
				holder_scope = {
					random_vassal = {
						limit = {
							among_most_powerful_vassals = 5
							NOT = { has_character_flag = is_fe4_edain }
							NOT = { has_character_flag = is_fe4_brigid }
							NOT = { has_character_flag = is_fe4_lex }
							NOT = { has_character_flag = is_fe4_tailtiu }
							NOT = { has_character_flag = is_fe4_ethnia }
							NOT = { has_character_flag = is_fe4_claud }
							NOT = { has_character_flag = is_fe4_alec }
							NOT = { has_character_flag = is_fe4_naoise }
							NOT = { has_character_flag = is_fe4_arden }
							NOT = { has_character_flag = is_fe4_jamke }
							NOT = { has_character_flag = is_fe4_arvis }
							NOT = { has_character_flag = refused_belhalla_war }
						}
						set_character_flag = battle_of_belhalla_war_leader
					}
				}
			}
		}
		random_playable_ruler = { 
			limit = { has_character_flag = battle_of_belhalla_war_leader }
			liege = {
				character_event = { id = fe4_belhalla.3 } 
			}
		}
	}

	id = fe4_belhalla.11
	hide_window = yes
	is_triggered_only = yes
	
	option = {
		name = OK
		if = {
			limit = {
				FROMFROM = {
					any_war = {
						using_cb = battle_of_belhalla
					}
				}
			}
			set_character_flag = yied_massacre_perpetrator
			# Did the wife come?
			if = {
				limit = {
					FROMFROM = {
						any_spouse = {
							has_character_flag = yied_massacre_target
						}
					}
				}
				FROMFROM = {
					any_spouse = {
						limit = { has_character_flag = yied_massacre_target }
						character_event = { id = fe4_belhalla.12 }
					}
				}
			}
			# If the wife is not around, check to see if she brought the kid before leaving
			else_if = {
				limit = {
					FROMFROM = {
						any_child = {
							is_adult = no
							prisoner = no
							has_character_flag = brought_to_yied
						}
					}				
				}
				FROMFROM = {
					random_child = {
						limit = {
							is_adult = no
							prisoner = no
							has_character_flag = brought_to_yied
						}
						character_event = { id = fe4_belhalla.13 }
					}
				}		
				
			}
			# And kill Quan
			FROMFROM = {
				character_event = { id = fe4_belhalla.15 days = 1 }
			}
		}		
	}		
}
