namespace = galle

# Galle conquers a county
character_event = {
	id = galle.1
	desc = galle.1.desc
	picture = GFX_evt_heretic
	is_triggered_only = yes
	hide_from = yes
	
	option = {
		name = galle.1.A
		ai_chance = {
			factor = 1
		}
		set_character_flag = galle_county_conquest_allowed
		# Find a target. 
		# Pick a random neighbor.
		random_neighbor_independent_ruler = {
			limit = {
				NOT = { has_landed_title = k_grann }  
			}
			# Pick a province in their realm that borders one of ours
			random_realm_province = {
				limit = {
					any_neighbor_province = {
						holder_scope = {
							OR = {
								character = ROOT
								top_liege = { character = ROOT }
							}
						}
					}
				}
				county = {
					holder_scope = {
						top_liege = {
							reverse_war = {
								target = ROOT
								casus_belli = galle_county_conquest
								thirdparty_title = PREVPREV
							}
						}
					}
				}
			}
		}
		spawn_galle_stack_effect = yes
	}	
	option = {
		name = galle.1.B
		ai_chance = {
			factor = 0
		}
		d_loptyr = {
			destroy_landed_title = THIS
		}
	}	
}

# The Great Purge
character_event = {
	id = galle.4
	desc = galle.4.desc
	picture = GFX_evt_shadowy_cabal
	is_triggered_only = yes
	hide_from = yes
	
	only_playable = yes
	religion = loptyrian
	has_global_flag = grann_republic_destroyed
	trigger = {
		year = 449
		NOT = { year = 457 }
		NOT = { has_global_flag = great_purge }
		imperial_loptyr_in_power_trigger = yes
		any_vassal = {
			NOT = { religion = loptyrian }
		}
	}
	
	immediate = {
		set_global_flag = great_purge
	}
	
	# Purge the dissidents!
	option = {
		name = galle.4.A
		ai_chance = {
			factor = 1
		}
		prestige = 500
		custom_tooltip = {
			text = galle.4.A.tooltip
			hidden_tooltip = {
				any_vassal = {
					limit = {
						NOT = { religion = loptyrian }
					}
					capital_scope = {
						add_province_modifier = { name = great_purge duration = 365 }
					}
					if = {
						limit = { ai = yes }
						any_demesne_title = {
							usurp_title = { target = ROOT type = claim }
						}
						death = { 
							death_reason = death_execution
							killer = ROOT
						} 
					}else = {
						any_demesne_title = {
							limit = { higher_tier_than = COUNT }
							usurp_title = { target = ROOT type = claim }
						}
						imprison = ROOT
					}
				}
			}
		}
		
		hidden_tooltip = {
			any_playable_ruler = {
				if = {
					limit = {
						primary_title = { region = world_jugdral }
						is_within_diplo_range = ROOT
					}
					narrative_event = { id = galle.5 days = 1 }
				}
			}
		}
	}	
	# Nah, I'll ignore them
	option = {
		name = galle.4.B
		ai_chance = {
			factor = 0
		}
		any_vassal = {
			limit = {
				NOT = { religion = loptyrian }
			}
			opinion = {
				modifier = opinion_evil_tyrant
				who = ROOT
				years = 10
			}
		}
	}	
}
