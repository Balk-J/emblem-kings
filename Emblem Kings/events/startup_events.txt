namespace = startup

character_event = {
	id = startup.1
	#desc = "startup.1.desc"
	
	is_triggered_only = yes #Triggered from on_startup
	only_rulers = yes
	
	trigger = {
		NOT = { has_global_flag = initialized }
		has_landed_title = k_naga
	}
	immediate = { 
		set_global_flag = initialized 
	}
	
	option = {
		#name = startup.1.A
		name = OK
		# Fix automaticaally created barons with wrong culture/religion
		any_landed_title = {
			limit = {
				tier = BARON
			}
			holder_scope = {
				if = {
					limit = {
						tier = BARON
						NOT = {
							liege = {
								religion = PREV
							}
						}
						NOT = {
							liege = {
								culture = PREV
							}
						}
					}
					character_event = { id = util.4 }
				}
			}
		}
			
		if = {
			limit = { year = 440 }
			# Structures on Valor
			1 = { #Virtus
				add_province_modifier = { name = ancient_ruins duration = -1 }
				add_province_modifier = { name = dragons_gate_elibe duration = -1 }
			}
			2 = { #Temperantia
				add_province_modifier = { name = ancient_ruins duration = -1 }
			}
			# The area in Ilia where Bern found that weather magic tome
			261 = { #Neandreia
				add_province_modifier = { name = ancient_ruins duration = -1 }
			}
			# Arcadia
			116 = { 
				add_province_modifier = { name = ancient_ruins duration = -1 }
			}
		}
		
		# Artifacts
		character_event = { id = startup_artifacts.1 }
		
		# Create immortal characters via event, so they look the appropriate age
		if = { # Athos and Bramimond
			limit = { 
				year = 120
				NOT = { year = 1161 }
			}
			character_event = {
				id = fe_immortal.1
			}
		}
		if = { # Ninian and Nils
			limit = { 
				year = 1159
				NOT = { year = 1161 }
			}
			character_event = {
				id = fe_immortal.2
			}
		}
		if = { # Fae
			limit = { 
				year = 845
				NOT = { year = 4204 }
			}
			character_event = {
				id = fe_immortal.3
			}
		}
		if = { # Sophia
			limit = { 
				year = 905
				NOT = { year = 1400 }
			}
			character_event = {
				id = fe_immortal.4
			}
		}
		
		# Jugdral-specific character stuff
		character_event = {
			id = startup.5
		}
		
		# Startup events for scenarios
		if = { # Genealogy
			limit = { 
				year = 755
				NOT = { year = 756 }
			}
			character_event = {
				id = startup.9
			}
		}
		if = { # Genealogy
			limit = { 
				year = 756
				NOT = { year = 760 }
			}
			character_event = {
				id = startup.8
			}
		}
		if = { # Lyn's Story
			limit = { 
				year = 1159
				NOT = { year = 1160 }
			}
			character_event = {
				id = startup.3
			}
		}
		
		if = { # Blazing Sword
			limit = { 
				year = 1160
				NOT = { year = 1161 }
			}
			character_event = {
				id =  startup.4
			}
		}
		
		if = { # Binding Blade
			limit = { 
				year = 1180
				NOT = { year = 1181 }
			}
			character_event = {
				id =  startup.2
			}
