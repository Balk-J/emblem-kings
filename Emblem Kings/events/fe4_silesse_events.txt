	id = fe4_silesse.5

		name = OK

			}

					death = yes
				}
			}

			limit = {
				d_orgahil_pirates = {
					holder_scope = {
						ai=yes

							has_character_flag=is_fe4_brigid
							AND = {
								current_heir = {
									ai = yes
									has_character_flag=is_fe4_brigid
								}
							}
						}
					}
				}
			}
			d_orgahil_pirates = {
				holder_scope = {
					random_vassal = {
						d_orgahil_pirates = {
							holder_scope = {
								any_demesne_title = {
									usurp_title = PREVPREVPREV
								}
							}
						}
					}
					character_event = { id = fe4_silesse.6 }
				}
			}
		}
		# Human Brigid should just get approached by Edain
		character_event = { id = fe4_silesse.7 days = 1}
	}		
}

							} 	
						}
					}
					
					c_sailane = {
						grant_title = ROOT
					}
					d_sailane = {
						grant_title = ROOT
					}
					
					any_demesne_title = {
						limit = {
							NOT = { title = d_sailane }
							NOT = { de_jure_liege_or_above = d_sailane }
						}
						usurp_title = FROM
					}
					k_silesse = {		
						holder_scope = {
							ROOT = {
								set_defacto_liege = PREV
							}
						}
					}
				}else = {
					# Sailane is not AI controlled or somehow got in the realm of Sigurd's liege.
					any_demesne_title = {
						usurp_title = FROM
					}	
					random_neighbor_independent_ruler = {
						limit = {
							any_demesne_title = { can_be_given_away = yes } 
						}
						random_demesne_title = {
							limit = { tier = COUNT can_be_given_away = yes }
							grant_title = ROOT
							ROOT = {
								set_defacto_liege = PREVPREV
							}
						}
					}
				}
			}
		}
	}
	option = {
		name = fe4_silesse.14.B
		ai_chance = {
			factor = 15
			modifier = {
				factor = 0

					value = on
				}
			}
		}			
		custom_tooltip = {
			text = fe4_silesse.14.B.tooltip
			FROM = {
				letter_event = { id = fe4_silesse.15	}
			}
		}
	}
